MONGODB + KUBERNETES COMMANDS CHEAT SHEET
=========================================

1) KUBERNETES - DEPLOY / UPDATE APP
-----------------------------------
# Apply all manifests from k8s/
kubectl apply -k k8s

# Update image in deployment (replace tag as needed)
kubectl -n mern set image deployment/mern-app mern-app=058264503680.dkr.ecr.ap-south-1.amazonaws.com/mern-app:v2

# Restart deployment
kubectl -n mern rollout restart deployment/mern-app

# Check rollout status
kubectl -n mern rollout status deployment/mern-app --timeout=300s

# Check pods/services
kubectl -n mern get pods -o wide
kubectl -n mern get svc
kubectl -n mern get all -o wide

# App logs
kubectl -n mern logs deployment/mern-app --tail=100


2) KUBERNETES - SERVICE ACCESS
------------------------------
# NodePort URL format (current setup)
# http://<worker-node-public-ip>:30050

# Get node public IPs
kubectl get nodes -o wide

# Health check endpoint
# http://<worker-node-public-ip>:30050/api/health

# Port-forward app service to local machine
kubectl -n mern port-forward svc/mern-app 5000:5000
# then open: http://localhost:5000/api/health


3) KUBERNETES - MONGO INSPECTION
--------------------------------
# Check Mongo pod/service/pvc
kubectl -n mern get pod -l app=mongo -o wide
kubectl -n mern get svc mongo
kubectl -n mern get pvc mongo-pvc -o wide

# Open Mongo shell inside Mongo deployment
kubectl -n mern exec -it deploy/mongo -- mongosh

# Run one-shot Mongo commands from kubectl (non-interactive)
kubectl -n mern exec deploy/mongo -- mongosh --quiet --eval "show dbs"
kubectl -n mern exec deploy/mongo -- mongosh --quiet --eval "db.getSiblingDB('mern_app').getCollectionNames()"
kubectl -n mern exec deploy/mongo -- mongosh --quiet --eval "db.getSiblingDB('mern_app').users.countDocuments()"
kubectl -n mern exec deploy/mongo -- mongosh --quiet --eval "db.getSiblingDB('mern_app').issues.countDocuments()"
kubectl -n mern exec deploy/mongo -- mongosh --quiet --eval "db.getSiblingDB('mern_app').users.find().limit(5).toArray()"
kubectl -n mern exec deploy/mongo -- mongosh --quiet --eval "db.getSiblingDB('mern_app').issues.find().limit(5).toArray()"


4) MONGODB (MONGOSH) COMMANDS
-----------------------------
# Connect (local/port-forwarded)
mongosh "mongodb://localhost:27017/mern_app"

# Basic navigation
show dbs
use mern_app
show collections

# Read data
db.users.find().pretty()
db.issues.find().pretty()
db.users.find().limit(5).pretty()
db.issues.countDocuments()

# Delete data
db.users.deleteOne({ _id: ObjectId("YOUR_ID") })
db.issues.deleteMany({ status: "Done" })
db.users.deleteMany({})
db.issues.drop()
db.dropDatabase()


5) DELETE / RESET DATABASE IN CLUSTER
-------------------------------------
# Drop only mern_app database
kubectl -n mern exec deploy/mongo -- mongosh --quiet --eval "db.getSiblingDB('mern_app').dropDatabase()"

# Full reset (delete mongo deployment + PVC, then recreate)
kubectl -n mern delete deployment mongo
kubectl -n mern delete pvc mongo-pvc
kubectl apply -k k8s


6) OPTIONAL - ECR BUILD/PUSH FLOW
---------------------------------
# Build Docker image (run from project root)
docker build -t 058264503680.dkr.ecr.ap-south-1.amazonaws.com/mern-app:v2 .

# Login to ECR
aws ecr get-login-password --region ap-south-1 | docker login --username AWS --password-stdin 058264503680.dkr.ecr.ap-south-1.amazonaws.com

# Push image
docker push 058264503680.dkr.ecr.ap-south-1.amazonaws.com/mern-app:v2

# Deploy pushed image
kubectl -n mern set image deployment/mern-app mern-app=058264503680.dkr.ecr.ap-south-1.amazonaws.com/mern-app:v2
kubectl -n mern rollout status deployment/mern-app --timeout=300s


7) TROUBLESHOOTING QUICK CHECKS
-------------------------------
# Confirm app is up
kubectl -n mern get pods -o wide
kubectl -n mern logs deployment/mern-app --tail=50

# Describe failing pods
kubectl -n mern describe pod -l app=mern-app
kubectl -n mern describe pod -l app=mongo

# Confirm app env var in deployment
kubectl -n mern get deploy mern-app -o jsonpath="{.spec.template.spec.containers[0].env[?(@.name=='MONGODB_URI')].value}"
